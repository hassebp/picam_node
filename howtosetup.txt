### Opsætning af ny picam node ###

# Klon projekt fra github
cd /home/pi
git clone https://github.com/hassebp/picam_node projekt 

cd /home/pi/projekt 
apt install npm
npm install -g nodemon
npm install

indsæt picam navn 
nano /etc/hostname

indsæt picam navn
nano /etc/hosts

sudo apt-get install -y libjasper-dev libjpeg8-dev liblcms2-dev
sudo apt-get install -y ffmpeg
sudo apt-get install -y git

git clone https://github.com/RobertElderSoftware/fork-raspiraw && cd fork-raspiraw && git checkout 18fac55136f98960ccd4dcfff95112134e5e45db

Kopier https://github.com/6by9/raspiraw/blob/master/camera_i2c ind i /home/pi/fork-raspiraw/camera_i2c

./buildme

git clone https://github.com/RobertElderSoftware/dcraw && cd dcraw && git checkout 8d2bcbe8f9d280a5db8da30af9b6eb034f7f2859
./buildme

sudo apt-get install wiringpi
sudo apt-get install i2c-tools

Indsæt dtparam=i2c_vc=on i /boot/config.txt
Indsæt i2c-dev i /etc/modules-load.d/modules.conf

Genstart

mkdir /opt/temp/

Kør crontab -e
indsæt:
@reboot cd /home/pi/projekt/ && sudo -u root nodemon index.js > /home/pi/projekt/index.log
